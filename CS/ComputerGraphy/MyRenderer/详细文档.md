uniform vec4 globalAmbient; //全局环境光
uniform PositionalLight light; 
uniform Material material;
uniform mat4 mv_matrix; //模型视图矩阵
uniform mat4 proj_matrix; //投影矩阵
uniform mat4 norm_matrix; //法线变换到视图空间的矩阵
